```latex
\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{positioning, shapes.geometric, arrows.meta}

\begin{document}
\begin{tikzpicture}[
    node distance=1.5cm,
    block/.style={rectangle, draw, fill=blue!20, rounded corners, text centered},
    arrow/.style={-Triangle, thick}
]

% 输入序列
\node[block] (input) {输入序列\\$[batch\_size, seq\_len]$};

% Embedding层
\node[block, right=of input] (embedding) {Embedding层\\$nn.Embedding(vocab\_size, embed\_dim)$\\输入: $[batch\_size, seq\_len]$\\输出: $[batch\_size, seq\_len, embed\_dim]$};

% CFC层
\node[block, right=of embedding] (cfc) {CFC层\\$Cfc(in\_features=embed\_dim, hidden\_size=hidden\_dim, out\_feature=hidden\_dim, hparams=hparams, return\_sequences=True)$\\输入: $[batch\_size, seq\_len, embed\_dim]$\\输出: $[batch\_size, seq\_len, hidden\_dim]$};

% 分类层
\node[block, right=of cfc] (classifier) {分类层\\$nn.Linear(hidden\_dim, output\_dim)$\\输入: $[batch\_size, seq\_len, hidden\_dim]$\\输出: $[batch\_size, seq\_len, output\_dim]$};

% 输出预测
\node[block, right=of classifier] (output) {输出预测\\$[batch\_size, seq\_len, output\_dim]$};

% 连接箭头
\draw[arrow] (input) -- (embedding);
\draw[arrow] (embedding) -- (cfc);
\draw[arrow] (cfc) -- (classifier);
\draw[arrow] (classifier) -- (output);

\end{tikzpicture}
\end{document}
```
